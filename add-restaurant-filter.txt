Backend API endpoint banao:
server/routes/sales.ts mein add karo:

```typescript
export const handleGetRestaurants: RequestHandler = async (req, res) => {
  try {
    const db = await getDatabase();
    const collection = db.collection("petpooja");
    
    const restaurants = await collection.distinct("restaurant_name");
    
    res.json({ success: true, restaurants: restaurants.filter(Boolean).sort() });
  } catch (error) {
    console.error("Failed to fetch restaurants:", error);
    res.status(500).json({ success: false, error: "Failed to fetch restaurants" });
  }
};
```

server/index.ts mein route add karo:
```typescript
app.get("/api/sales/restaurants", handleGetRestaurants);
```
